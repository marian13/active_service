def method_missing(method, value = nil)
  key = method.to_s

  if method.end_with?("=")
    define_singleton_method(method) { |value| @hash.store(key.chomp("="), value) }

    send(method, value)
  else
    define_singleton_method(method) { @hash.fetch(key) { @hash.store(key, self.class.new) } }

    send(method)
  end
end
