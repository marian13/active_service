def copy(overrides: {})
  overrides[:args] ||= {}
  overrides[:kwargs] ||= {}

  args =
    if respond_to?(:to_args)
      case overrides[:args]
      when ::Array
        overrides[:args]
      when ::Hash
        Utils::Array.merge(to_args, overrides[:args])
      end
    else
      []
    end

  kwargs =
    if respond_to?(:to_kwargs)
      to_kwargs.merge(overrides[:kwargs])
    else
      {}
    end

  block =
    if respond_to?(:to_block)
      overrides[:block] || to_block
    end

  self.class.new(*args, **kwargs, &block)
end
